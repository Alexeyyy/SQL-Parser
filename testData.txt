SELECT f.aircraft_code AS distinct
FROM flights AS f
	INNER JOIN aircrafts AS a ON  not a.aircraft_code=f.aircraft_code
WHERE (((not (f.flight_id>10 and f.flight_id<20) and f.flight_id not between (SELECT 11 ) and (SELECT 20 )) and f.flight_id in (1,3,4,(SELECT DISTINCT ff.flight_id FROM flights AS ff LIMIT 1 ))) and f.aircraft_code like '%4343%')
LIMIT 10
