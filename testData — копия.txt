select data.city, count(*)
from (
    select big_air_landings.city, a.airport_name, a.airport_code
    from (
             select ap.city as city, count(*) as ap_count
             from airports as ap
             group by ap.city
             having count(*) > 1
         ) as big_air_landings
        join airports as a on a.city = big_air_landings.city
) as data
where (1=1 or 2=2 and 3=3 or (5=5 and 6=6 and ((7=7 or (18-5 = 13)))) and (((select 1 limit 1) = 1) and 4 between 1 and 4)) and data.city like '%лья%'
group by data.city
